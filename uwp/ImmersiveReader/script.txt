<body onload="handleLaunchImmersiveReader()">

    <script type='text/javascript'>

        async function handleLaunchImmersiveReader() {
            const data = {
                title: ' ',
                chunks: [ {
                    content: '|CONTENT|',
                    lang: 'en'
                } ]
            };

            const options = {
                uiZIndex: 1000000,
                hideExitButton: true,
                allowFullscreen: false
            }

            ImmersiveReader.launchAsync('|TOKEN|', '|YOUR_SUB_DOMAIN|', data, options)
                .then(() => {
                    console.log('success');
                }, (error) => {
                    console.log('error! ' + error.message + ' ' + error.name);
                });
        }
    </script>
</body>
